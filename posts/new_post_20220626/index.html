#!/usr/bin/python3

import requests
import json


class Auto_ops:
    # attribute
    api_token = ""
    origin_url = "http://localhost:10445"
    headers = {
        'Content-type': 'application/json',
        'Accept': 'application/json',
        "Authorization": api_token
    }
    id_project = ""

    # Constructor
    def __init__(self, origin_url):
        self.api_token = self.set_conf()
        self.origin_url = origin_url

    def set_conf(self):

        # Opening JSON file
        file = open('configmap.json')

        # returns JSON object as a dictionary
        data = json.load(file)
        api_token = "Bearer" + " " + data["token"]
        self.headers["Authorization"] = api_token

        return api_token

    def show_conf(self):
        print(self.api_token)
        print(self.headers)

    def search_project_id(self):
        url = self.origin_url + "/api/v2/projects/"
        response = requests.get(url=url, headers=self.headers)

        if(response.status_code == 200):

            json_object = json.loads(
                response.content.decode('utf8').replace("'", '"'))

            print(json.dumps(json_object, indent=4, sort_keys=True))
            print('- ' * 20)
            print(type(json_object))
            # print( json_object["results"][0]["id"] )
            for j in json_object["results"]:
                if j["name"] == "TEST":
                    print(j["id"])


ops = Auto_ops("http://localhost:10445")
ops.show_conf()
ops.search_project_id()
